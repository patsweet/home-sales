def scatterPlusTrend(xval, yval):
    """Create a scatterplot with a trendline using
    scipy's linregress function to perform the least
    squares regression."""
    plt.figure(figsize=(15,8))
    plt.grid()
    slope, intercept, r_value, p_value, std_err = linregress(x=df[xval], y=df[yval])
    print 'R-Squared: {:.5f}'.format(r_value**2)
    print 'P Value: {:.5f}'.format(p_value)
    print 'Standard Error: {:,.2f}'.format(std_err)
    plt.scatter(x=df[xval], y=df[yval])
    plt.plot(df[xval], slope * df[xval] + intercept, 'r-')

def histogramPlusAgg(var):
    """Produces a histrogram based on the provided variable
    along with very basic aggregate stats - mean, max and min -
    formatted to 4 decimal places."""
    print df[var].hist(bins=60,figsize=[15,8])
    print "Average {}: {:,.4f}".format(var, df[var].mean())
    print "Max {}: {:,.4f} ({!s})".format(
        var,
        df[var].max(),
        df[df[var] == df[var].max()].index[0]
    )
    print "Min {}: {:,.4f} ({})".format(
        var,
        df[var].min(),
        df[df[var] == df[var].min()].index[0]
    )

There doesn't seem to be too strong of a correlation between Days on Market and the Sale Price/Original List Price ratio. The least squares regression shows that the longer the home is on the market, the least likely the seller is to get the original listing price. However, the r-squared produced by the regression suggests the relationship is not very strong.
